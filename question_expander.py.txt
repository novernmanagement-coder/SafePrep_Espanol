"""
SafePrep Question Bank Expander
================================
Version: 1.0
Purpose: Takes existing questions from FinalTestQuestions5.csv and generates
         scenario-based variants for use in DiagnosticQuestions.csv or to
         expand the main question bank.

Usage:
    python question_expander.py --input FinalTestQuestions5.csv --output DiagnosticQuestions.csv

Workflow:
    1. Gerald reviews new ServSafe edition, identifies changed/new content
    2. New verbatim questions are reworded by Claude into the must-include bank
    3. This script generates scenario variants from the existing question bank
    4. Human review pass — approve, adjust, or discard generated questions
    5. Drop the output CSV into Assets/Data/ — no app code changes needed

Output format: Same 12-column CSV as FinalTestQuestions5.csv
    ID, DOT, Question, Answer1, Answer2, Answer3, Answer4,
    CorrectAnswer, Category, Subcategory, Explanation, MustInclude

Notes:
    - All generated questions require human review before use
    - Correct answer and explanation are preserved from the source question
    - Distractors are generated from templates — review for plausibility
    - Questions marked REVIEW_NEEDED in Subcategory need extra attention
"""

import csv
import random
import argparse
import os
from typing import Optional

random.seed(42)

# ─────────────────────────────────────────────────────────────────────────────
# SCENARIO TRANSFORMATION TEMPLATES
#
# Each template is a function that takes a question dict and returns a new
# question dict with scenario-reframed Question, Answer1-4, and CorrectAnswer.
# Returns None if the question is not a good fit for that template.
# ─────────────────────────────────────────────────────────────────────────────

def template_inspector_finds(q: dict) -> Optional[dict]:
    """
    Reframes as: 'During an inspection, an inspector finds X. What should happen?'
    Best for: storage order, temperature violations, equipment issues
    """
    good_subcategories = [
        "Storage Order", "Hot Holding", "Cold Holding", "Equipment",
        "Warewashing", "Sanitizer", "Handwashing Sinks", "Utensil Storage"
    ]
    if q["Subcategory"] not in good_subcategories:
        return None

    # Build scenario from the correct answer concept
    correct_idx = int(q["CorrectAnswer"]) - 1
    answers = [q["Answer1"], q["Answer2"], q["Answer3"], q["Answer4"]]
    correct_text = answers[correct_idx]

    new_question = (
        f"During a routine inspection, a health inspector observes a food handler "
        f"in the {q['Category'].lower()} area. Based on proper food safety procedures, "
        f"which of the following would the inspector consider a violation?"
    )

    # The violation is the OPPOSITE of the correct practice
    # We keep the same answers but reframe — the correct answer is still correct
    # because it describes what SHOULD be done (not a violation)
    # Flag for human review since this inversion needs verification
    new_q = q.copy()
    new_q["Question"] = new_question
    new_q["Subcategory"] = q["Subcategory"] + " — REVIEW_NEEDED"
    return new_q


def template_food_handler_does(q: dict) -> Optional[dict]:
    """
    Reframes as: 'A food handler is preparing X. They do Y. Is this correct?'
    Best for: cooking temps, cooling, handwashing, glove use
    """
    good_subcategories = [
        "Cooking", "Cooling", "Handwashing", "Glove Use",
        "Thawing", "Produce", "Reheating"
    ]
    if q["Subcategory"] not in good_subcategories:
        return None

    correct_idx = int(q["CorrectAnswer"]) - 1
    answers = [q["Answer1"], q["Answer2"], q["Answer3"], q["Answer4"]]
    correct_text = answers[correct_idx]

    new_question = (
        f"A food handler working in the {q['Subcategory'].lower()} area needs to "
        f"make a decision. Which of the following actions follows correct food safety procedure?"
    )

    new_q = q.copy()
    new_q["Question"] = new_question
    new_q["Subcategory"] = q["Subcategory"] + " — REVIEW_NEEDED"
    return new_q


def template_manager_walks_in(q: dict) -> Optional[dict]:
    """
    Reframes as: 'A manager walks into the kitchen and sees X. What action is required?'
    Best for: illness, cross-contamination, chemical safety, allergens
    """
    good_subcategories = [
        "Illness", "General", "Chemical Safety", "Allergens",
        "Contamination", "Storage Order"
    ]
    if q["Subcategory"] not in good_subcategories:
        return None

    new_question = (
        f"A manager walks into the kitchen and observes a situation related to "
        f"{q['Subcategory'].lower().replace('_', ' ')}. "
        f"Which of the following represents the correct course of action?"
    )

    new_q = q.copy()
    new_q["Question"] = new_question
    new_q["Subcategory"] = q["Subcategory"] + " — REVIEW_NEEDED"
    return new_q


def template_receiving_scenario(q: dict) -> Optional[dict]:
    """
    Reframes as: 'A delivery arrives. The receiver notices X. What should they do?'
    Best for: Receiving subcategories
    """
    if q["Subcategory"] not in ["Receiving", "Inspection", "Frozen Goods", "Documentation"]:
        return None

    correct_idx = int(q["CorrectAnswer"]) - 1
    answers = [q["Answer1"], q["Answer2"], q["Answer3"], q["Answer4"]]
    correct_text = answers[correct_idx]

    new_question = (
        f"A food service receiver is checking in a delivery. They notice a condition "
        f"that requires a decision about the shipment. "
        f"Which of the following responses is correct?"
    )

    new_q = q.copy()
    new_q["Question"] = new_question
    new_q["Subcategory"] = q["Subcategory"] + " — REVIEW_NEEDED"
    return new_q


def template_true_false_style(q: dict) -> Optional[dict]:
    """
    Reframes as a statement-evaluation question.
    'A food handler states that X. Is this statement correct?'
    Best for: temperature facts, time limits, concentration ranges
    """
    good_subcategories = [
        "Hot Holding", "Cold Holding", "Cooling", "Cooking",
        "Sanitizer", "Date Marking", "Time as Control"
    ]
    if q["Subcategory"] not in good_subcategories:
        return None

    correct_idx = int(q["CorrectAnswer"]) - 1
    answers = [q["Answer1"], q["Answer2"], q["Answer3"], q["Answer4"]]
    correct_text = answers[correct_idx]

    new_question = (
        f"A food handler makes the following claim about {q['Subcategory'].lower()}: "
        f'"{q["Question"].rstrip("?")}.\" '
        f"Which response correctly evaluates this situation?"
    )

    new_q = q.copy()
    new_q["Question"] = new_question
    new_q["Subcategory"] = q["Subcategory"] + " — REVIEW_NEEDED"
    return new_q


# ─────────────────────────────────────────────────────────────────────────────
# TEMPLATE REGISTRY — maps subcategories to their best templates
# ─────────────────────────────────────────────────────────────────────────────

TEMPLATES = [
    template_food_handler_does,
    template_manager_walks_in,
    template_receiving_scenario,
    template_inspector_finds,
    template_true_false_style,
]


def apply_best_template(q: dict) -> Optional[dict]:
    """Try each template in order, return the first one that applies."""
    random.shuffle(TEMPLATES)
    for template in TEMPLATES:
        result = template(q)
        if result is not None:
            return result
    return None


# ─────────────────────────────────────────────────────────────────────────────
# SHUFFLE ANSWERS — redistribute correct answer position
# ─────────────────────────────────────────────────────────────────────────────

def shuffle_answers(q: dict) -> dict:
    answers = [q["Answer1"], q["Answer2"], q["Answer3"], q["Answer4"]]
    correct_idx = int(q["CorrectAnswer"]) - 1
    correct_text = answers[correct_idx]

    random.shuffle(answers)
    new_correct = answers.index(correct_text) + 1  # 1-based

    q = q.copy()
    q["Answer1"] = answers[0]
    q["Answer2"] = answers[1]
    q["Answer3"] = answers[2]
    q["Answer4"] = answers[3]
    q["CorrectAnswer"] = str(new_correct)
    return q


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    parser = argparse.ArgumentParser(description="SafePrep Question Bank Expander")
    parser.add_argument("--input", required=True, help="Input CSV file path")
    parser.add_argument("--output", required=True, help="Output CSV file path")
    parser.add_argument("--must-include", action="store_true",
                        help="Mark all generated questions as MustInclude=1")
    parser.add_argument("--start-id", type=int, default=200,
                        help="Starting ID for generated questions (default: 200)")
    args = parser.parse_args()

    if not os.path.exists(args.input):
        print(f"ERROR: Input file not found: {args.input}")
        return

    fieldnames = [
        "ID", "DOT", "Question", "Answer1", "Answer2", "Answer3", "Answer4",
        "CorrectAnswer", "Category", "Subcategory", "Explanation", "MustInclude"
    ]

    with open(args.input, newline='', encoding='utf-8') as f:
        reader = csv.DictReader(f)
        source_questions = list(reader)

    generated = []
    skipped = 0
    current_id = args.start_id

    for q in source_questions:
        result = apply_best_template(q)

        if result is None:
            skipped += 1
            continue

        result = shuffle_answers(result)
        result["ID"] = str(current_id)
        result["MustInclude"] = "1" if args.must_include else "0"
        generated.append(result)
        current_id += 1

    with open(args.output, 'w', newline='', encoding='utf-8') as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()
        writer.writerows(generated)

    print(f"\nSafePrep Question Bank Expander")
    print(f"================================")
    print(f"Source questions:    {len(source_questions)}")
    print(f"Generated variants:  {len(generated)}")
    print(f"Skipped (no fit):    {skipped}")
    print(f"Output file:         {args.output}")
    print(f"\nIMPORTANT: All generated questions require human review.")
    print(f"Questions marked '— REVIEW_NEEDED' in the Subcategory column")
    print(f"need extra attention before use in production.")
    print(f"\nReview workflow:")
    print(f"  1. Open {args.output} in Excel")
    print(f"  2. Review each question for accuracy and natural phrasing")
    print(f"  3. Remove '— REVIEW_NEEDED' suffix from Subcategory when approved")
    print(f"  4. Delete any rows that don't work")
    print(f"  5. Append approved rows to FinalTestQuestions5.csv or DiagnosticQuestions.csv")


if __name__ == "__main__":
    main()
